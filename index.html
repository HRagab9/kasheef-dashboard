① كود التتبع — ضعه في كل صفحة
في <head> أو قبل </body> مباشرة
/* ── Kasheef Tracking Script ── */
<script>
window.KasheefTracker = {
  siteId: 'kasheef_live_2025',
  apiUrl: 'https://kasheef-server-production.up.railway.app/api/track',

  init() {
    this.trackPageView();
    this.trackForms();
    this.trackScrollDepth();
  },

  trackPageView() {
    this.send({ event: 'PageView',
      page: location.pathname,
      ref: document.referrer });
  },

  trackForms() {
    document.querySelectorAll('form')
      .forEach(f => f.addEventListener('submit', e => {
        const data = new FormData(e.target);
        this.send({ event: 'Lead',
          name: data.get('name'),
          phone: data.get('phone'),
          email: data.get('email') });
    }));
  },

  send(payload) {
    fetch(this.apiUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json',
                 'X-Site-Id': this.siteId },
      body: JSON.stringify(payload)
    });
  }
};
window.KasheefTracker.init();
</script>

/* أضف هذا بعد إرسال الفورم */
async function sendLeadToKasheef(formData) {
  const lead = {
    name:    formData.name,
    phone:   formData.phone,
    email:   formData.email,
    area:    formData.area,
    goal:    formData.goal,
    budget:  formData.budget,
    source:  'website',
    page:    location.href
  };

  await fetch('https://kasheef-server-production.up.railway.app/api/leads', {
    method: 'POST',
    headers: { 'Authorization': 'Bearer YOUR_API_KEY',
               'Content-Type': 'application/json' },
    body: JSON.stringify(lead)
  });
}
